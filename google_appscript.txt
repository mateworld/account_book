/**
 * 웹개발자용 가계부 API (오픈런가계부)
 */

function doGet(e) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName('Data');
  if (!sheet) return returnJson({ status: "error", message: "시트를 찾을 수 없음" });

  // 1. 검색어 보정: "2026-05" -> "202605"
  let monthFilter = (e.parameter.month || "").toString().replace(/[^0-9]/g, "");

  const data = sheet.getDataRange().getValues(); // 날짜 객체로 가져오기
  const headers = data[0]; 
  const rows = data.slice(1);

  const filteredRows = rows.filter(row => {
    if (!monthFilter) return true; 
    
    let dateVal = row[0]; // B열 (날짜)
    
    // 데이터가 날짜 객체가 아닐 경우 날짜 객체로 변환 시도
    if (!(dateVal instanceof Date)) {
      dateVal = new Date(dateVal);
    }
    
    // 유효한 날짜인지 확인
    if (isNaN(dateVal.getTime())) return false;

    // 2. [핵심] 시트의 날짜를 "202605" 형태의 문자열로 변환
    // monthFilter가 "2026"이면 "yyyy"와 비교, "202605"면 "yyyyMM"과 비교
    let format = monthFilter.length <= 4 ? "yyyy" : "yyyyMM";
    let sheetDateStr = Utilities.formatDate(dateVal, "GMT+9", format);
    
    // 3. 문자열 비교
    return sheetDateStr === monthFilter;
  });

  const jsonData = filteredRows.map(row => {
    let obj = {};
    headers.forEach((header, i) => {
      let val = row[i];
      // JSON 출력 시 날짜 형식을 예쁘게 유지하기 위함
      if (val instanceof Date) {
        val = Utilities.formatDate(val, "GMT+9", (header.toString().trim() === '날짜' ? "yyyy-MM-dd" : "yyyy-MM-dd HH:mm:ss"));
      }
      obj[header.toString().trim()] = val;
    });
    return obj;
  });

  return ContentService.createTextOutput(JSON.stringify(jsonData)).setMimeType(ContentService.MimeType.JSON);
}

// 응답용 함수
function returnJson(obj) {
  return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(ContentService.MimeType.JSON);
}


function doPost(e) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName('Data');
  
  try {
    const params = JSON.parse(e.postData.contents);
    
    // --- [삭제 로직 보정] ---
    if (params.action === 'delete') {
      const allData = sheet.getDataRange().getValues();
      let targetRow = -1;
      
      // 요청받은 날짜에서 시간 정보(T00:00...)를 제거하고 'YYYY-MM-DD'만 추출
      const requestDate = params.date.split('T')[0];

      for (let i = 1; i < allData.length; i++) {
        const rowDate = Utilities.formatDate(new Date(allData[i][0]), "GMT+9", "yyyy-MM-dd");
        const rowVendor = String(allData[i][1]).trim();
        const rowAmount = Number(allData[i][3]);

        if (rowDate === requestDate && 
            rowVendor === String(params.vendor).trim() && 
            rowAmount === Number(params.amount)) {
          targetRow = i + 1; 
          break;
        }
      }
      
      if (targetRow !== -1) {
        sheet.deleteRow(targetRow);
        return ContentService.createTextOutput(JSON.stringify({ "status": "success" })).setMimeType(ContentService.MimeType.JSON);
      } else {
        return ContentService.createTextOutput(JSON.stringify({ 
          "status": "error", 
          "message": "매칭 실패. (요청날짜: " + requestDate + ")" 
        })).setMimeType(ContentService.MimeType.JSON);
      }
    }

    // --- [입력 로직] ---
    const inputDate = params.date ? new Date(params.date) : new Date();
    sheet.appendRow([inputDate, params.vendor, params.content, Number(params.amount) || 0, params.account || "기타", params.type || "지출"]);
    return ContentService.createTextOutput(JSON.stringify({ "status": "success" })).setMimeType(ContentService.MimeType.JSON);
      
  } catch (err) {
    return ContentService.createTextOutput(JSON.stringify({ "status": "error", "message": err.toString() })).setMimeType(ContentService.MimeType.JSON);
  }
}
